%% ═══════════════════════════════════════════════════════════════
%% ROW 1  –  Data → Increments → Empirical PDF
%% ═══════════════════════════════════════════════════════════════

\node[raw, text width=2.2cm, minimum height=1.1cm] (raw)
  {\textbf{Raw Prices}\\$P(t)$\\[3pt]
   \scriptsize 1-min OHLCV\\8 futures markets};

\node[proc, text width=2.4cm, right=1.0cm of raw] (scale)
  {\textbf{Tick Scaling}\\[3pt]
   $P_{\mathbb{Z}}(t) = \lfloor P(t)/\Delta \rceil$\\[2pt]
   \scriptsize 1 tick $= 1$ integer unit};

\draw[arr] (raw) -- (scale) node[midway,above,lbl]{$\times 1/\Delta$};

\node[proc, text width=2.4cm, right=1.0cm of scale] (locf)
  {\textbf{Gap-Fill}\\[3pt]
   \scriptsize LOCF on missing\\1-min bars\\(zero-change rule)};

\draw[arr] (scale) -- (locf);

\node[proc, text width=3.2cm, right=1.0cm of locf] (incr)
  {\textbf{Lag Increments}\\[4pt]
   $\delta_\tau(t) = P(t) - P(t{-}\tau)$\\[4pt]
   \begin{tabular}{c|l}
     $\tau$ & grid\\\hline
     coarse & $N_\tau=20$\\
     fine   & $N_\tau=37$
   \end{tabular}};

\draw[arr] (locf) -- (incr);

\node[proc, text width=3.4cm, right=1.1cm of incr] (pdf)
  {\textbf{Empirical PDF}\\[4pt]
   $\hat{P}(x_j,\tau)=\dfrac{\#\{i:\delta_\tau(t_i)\in\text{bin }j\}}{N_\tau\cdot\Delta x}$\\[4pt]
   \scriptsize bin width $\Delta x = \varepsilon\cdot\Delta$,\;$\varepsilon\in\{0.1,\ldots,0.5\}$};

\draw[arr] (incr) -- (pdf);

%% ── Seasonality branch (below incr, feeds back as ΔP̃) ──────────────────────
\node[proc, text width=2.8cm, below=0.9cm of incr] (season)
  {\textbf{Seasonality Correction}\\[3pt]
   $\hat\sigma_m = \tfrac{1}{D_m}\sum|\delta_k|$\\[2pt]
   $\tilde\delta_\tau = \delta_\tau/\hat\sigma_{m}$};

\draw[arr] (incr.south) -- (season.north)
  node[midway,right,lbl]{optional};
\draw[arr] (season.east) -| (pdf.south)
  node[near start,below,lbl]{$\Delta\tilde{P}_\tau$};

%% ── ① Tip-scaling OLS ────────────────────────────────────────
\node[est, text width=3.6cm, above right=1.1cm and 0.1cm of fan] (tip)
  {\textbf{(1) Tip-Scaling OLS}\\[4pt]
   Extract centre bin $\hat{P}(0,\tau)$\\[2pt]
   $\log\hat{P}(0,\tau) = m\log\tau + c$\\[4pt]
   \begin{tabular}{c|c}

     0.00 & $-$2.14\\
     1.10 & $-$2.76\\
     $\vdots$ & $\vdots$
   \end{tabular}\\[4pt]
   OLS slope $\hat{m}$,\quad $R^2>0.99$};

\node[res, right=0.8cm of tip] (rtip) {$\hat\alpha_{\mathrm{tip}}=-1/\hat{m}$};

%% ── ② Collapse-quality minimiser ─────────────────────────────
\node[est, text width=3.6cm, right=0.1cm of fan] (coll)
  {\textbf{(2) PDF Collapse}\\[4pt]
   Rescale:\;$\tilde{x}=x/\tau^{1/\alpha}$,\;$\tilde{P}=\tau^{1/\alpha}\hat{P}$\\[3pt]
   $\mathrm{Score}(\alpha)=\!\dfrac{1}{|\mathcal{G}||\mathcal{T}|}
     \!\sum\![\log\tilde{P}-\mu_j]^2$\\[4pt]
   grid search $\alpha\in[1.2,2.8]$,\;160 pts};

\draw[arr] (fan) -- (coll.west);

\node[res, right=0.8cm of coll] (rcoll)
  {$\hat\alpha_{\mathrm{collapse}}=\arg\min\,\mathrm{Score}$};
\draw[arr] (coll) -- (rcoll);

%% ── ③ Structure functions ─────────────────────────────────────
\node[est, text width=3.6cm, below right=1.1cm and 0.1cm of fan] (sf)
  {\textbf{(3) Structure Functions}\\[4pt]
   $\hat{S}_q(\tau)=\tfrac{1}{N_\tau}\sum|\delta_\tau|^q$,\;
   $q\in\{0.25,\ldots,1.5\}$\\[3pt]
   log-log slope: $\hat\zeta(q)=q/\hat\alpha_q$\\[4pt]
   \begin{tabular}{c|c}
     $q$ & $\hat\alpha_q$\\\hline
     0.25 & 1.76\\
     0.75 & 1.79\\
     $\vdots$ & $\vdots$
   \end{tabular}};

\draw[arr] (fan) |- (sf.west);

\node[res, right=0.8cm of sf] (rsf)
  {$\hat\alpha_{\mathrm{sf}}=\mathrm{median}(\hat\alpha_q)$};
\draw[arr] (sf) -- (rsf);

%% ROW 3  –  Bias correction (below, violet dashed)
%% ═══════════════════════════════════════════════════════════════

\node[bias, text width=3.2cm, below=2.2cm of pdf] (rho)
  {\textbf{Autocorrelations}\\[3pt]
   $\hat\rho(k)=\mathrm{Corr}(\delta_t,\delta_{t-k})$\\[2pt]
   \scriptsize $k=1,\ldots,K_{\max}=60$};

\node[bias, text width=3.4cm, right=0.9cm of rho] (vr)
  {\textbf{Variance Ratio}\\[3pt]
   $\widehat{VR}(\tau)=1+2\!\sum_{k=1}^{\tau-1}\!\bigl(1-\tfrac{k}{\tau}\bigr)\hat\rho(k)$};

\node[bias, text width=2.8cm, right=0.9cm of vr] (bt)
  {\textbf{Bias Term}\\[3pt]
   $b(\tau)=-\tfrac{1}{2}\log\widehat{VR}(\tau)$};

\node[bias, text width=3.0cm, right=0.9cm of bt] (ystar)
  {\textbf{Corrected Response}\\[3pt]
   $y^*(\tau)=\log\hat{P}(0,\tau)-b(\tau)$\\[2pt]
   \scriptsize coefficient of $b$ fixed at 1};

\node[res, fill=orange!20, right=0.9cm of ystar] (rcorr)
  {$\hat\alpha_{\mathrm{corr}}=-1/\hat{m}^*$};

\caption{Full estimation pipeline.
\textbf{Top row}: raw 1-minute prices undergo tick scaling, LOCF gap-filling,
and lag-$\tau$ differencing to produce increments $\delta_\tau(t)$, from which
the empirical PDF $\hat{P}(x_j,\tau)$ is constructed at each lag on two
log-spaced grids ($N_\tau\in\{20,37\}$) and five bin-width values
($\varepsilon\in\{0.1,\ldots,0.5\}$).
An optional intraday seasonality correction replaces raw increments with
de-seasonalized increments $\tilde\delta_\tau$.
\textbf{Middle}: three independent estimators extract the Lévy exponent
from $\hat{P}$---tip-scaling OLS on the centre bin, collapse-quality
minimisation over a grid of 160 candidate $\alpha$ values, and
structure-function log-log regressions at six moment orders~$q$.
\textbf{Bottom row} (violet, dashed): lag-$k$ autocorrelations $\hat\rho(k)$
from the 1-minute series yield the variance ratio $\widehat{VR}(\tau)$,
whose log supplies the bias term $b(\tau)=-\tfrac{1}{2}\log\widehat{VR}(\tau)$.
Subtracting $b(\tau)$ from $\log\hat{P}(0,\tau)$ gives the corrected response
$y^*(\tau)$, and a constrained OLS refits the slope to obtain
$\hat\alpha_{\mathrm{corr}}$.}
